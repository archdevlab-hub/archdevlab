/** * ARCHITECT CORE ENGINE - v6.3 * "The Fiber Architect" - Smart State Diffing & Component Scheduler */ // 1. ADVANCED STATE MANAGER (With Safe ID Generation & Global Store) class Store { constructor(initialState) { this.state = initialState; this.listeners = new Set(); // Memory leak riskini azaltmak için Set kullanıyoruz } getState() { return this.state; } setState(nextState) { // Sığ karşılaştırma (Shallow comparison) - State gerçekten değişti mi? const hasChanged = Object.keys(nextState).some( key => this.state[key] !== nextState[key] ); if (!hasChanged) return; // Değişiklik yoksa render tetikleme! this.state = { ...this.state, ...nextState }; this.listeners.forEach(listener => listener(this.state)); localStorage.setItem('arch_fiber_state', JSON.stringify(this.state)); } // Unsubscribe (Abonelikten çıkma) fonksiyonu dönerek memory leak riskini sıfırlıyoruz subscribe(listener) { this.listeners.add(listener); return () => { this.listeners.delete(listener); }; } } // Güvenli ve çakışma ihtimali olmayan ID üretici (Date.now() açığını kapatır) const generateUniqueId = (() => { let count = 0; return (prefix = 'task') => { count++; return `${prefix}_${Date.now()}_${Math.random().toString(36).substr(2, 5)}_${count}`; }; })(); const savedState = localStorage.getItem('arch_fiber_state'); const AppStore = new Store(savedState ? JSON.parse(savedState) : { hours: 0, subject: "", tasks: [] }); // 2. DETACHED LOGIC ENGINE & NARRATIVE LAYER (Pure Functions Only) const EducationEngine = { generateTasks: function (hours, subject) { const p1 = Math.floor(hours * 0.3); const p2 = Math.floor(hours * 0.5); const p3 = hours - (p1 + p2); return [ { id: generateUniqueId('t1'), type: "📖 Temel Kurulum", text: `${subject} Konu Çalışması`, duration: p1, done: false }, { id: generateUniqueId('t2'), type: "🧠 Aktif Öğrenme", text: `Nitelikli Soru Çözümü`, duration: p2, done: false }, { id: generateUniqueId('t3'), type: "🔁 Kalıcı Öğrenme", text: `Hata Analizi & Geri Besleme`, duration: p3, done: false } ]; }, calculateProgress: function (tasks) { if (!tasks || tasks.length === 0) return 0; const doneCount = tasks.filter(t => t.done).length; return Math.round((doneCount / tasks.length) * 100); } }; const NarrativeManager = { getStrategyText: function (progress) { if (progress === 0) return "Döngüyü başlat 🔥 İlk adımı atmak korteksi tetikler."; if (progress > 0 && progress < 50) return "Süreç aktif 🚀 Derin odaklanmayı bozma, harika gidiyorsun."; if (progress >= 50 && progress < 100) return "Kritik eşik ⚡ Hatalarından öğrenme aşamasına geçiş yap."; return "Mükemmel yürütme 🎯 Bugünün nöral adaptasyon döngüsü başarıyla kapannedi!"; } }; // 3. SMART DIFFING COMPONENT SYSTEM (Gerçek Akıllı DOM Güncelleyici) const TaskComponent = { // String template yerine DOM düğümü (Node) üretiyoruz ki üzerinde diff yapabilelim createNode: function (task) { const template = document.createElement('div'); template.className = `task-card ${task.done ? 'task-status-done' : 'task-status-active'}`; template.setAttribute('data-id', task.id); template.innerHTML = `
${task.type}

${task.text}

⏱ Süre: ${task.duration} saat
`; return template; }, // SMART DIFF: Tüm kartı silip eklemek yerine sadece değişen sınıfları ve buton metnini günceller! updateNode: function (existingNode, task) { const isCurrentlyDone = existingNode.classList.contains('task-status-done'); // Eğer state ile DOM eşleşiyorsa hiçbir şey yapma (Gereksiz DOM manipülasyonu engellenir) if (isCurrentlyDone === task.done) return; // Sadece değişen sınıfları güncelle (Flicker/Göz kırpma sorununu çözer) if (task.done) { existingNode.classList.replace('task-status-active', 'task-status-done'); } else { existingNode.classList.replace('task-status-done', 'task-status-active'); } const button = existingNode.querySelector('.task-action-btn'); if (button) { button.className = `task-action-btn ${task.done ? 'btn-undo' : 'btn-complete'}`; button.innerHTML = task.done ? '🔄 Geri Al' : '✓ Tamamla'; } } }; // 4. APPLICATION SCHEDULER & CONTROLLER const EducationController = { init: function () { this.cacheDOM(); this.bindEvents(); // State'e abone ol ve gelen unsubscribe fonksiyonunu sakla (İleride memory cleanup için) this.unsubscribeStore = AppStore.subscribe((state) => this.reconcile(state)); this.reconcile(AppStore.getState()); }, cacheDOM: function () { this.taskListContainer = document.querySelector('#task-list'); this.progressBar = document.querySelector('#progress-bar-fill'); this.progressText = document.querySelector('#progress-title'); this.strategyText = document.querySelector('#strategy-text'); this.btnGenerate = document.querySelector('#btn-generate-plan'); }, bindEvents: function () { this.btnGenerate.addEventListener('click', () => { const hours = parseFloat(document.querySelector('#input-hours').value) || 0; const subject = document.querySelector('#input-subject').value.trim(); if (hours > 0 && subject) { const newTasks = EducationEngine.generateTasks(hours, subject); AppStore.setState({ hours, subject, tasks: newTasks }); } }); this.taskListContainer.addEventListener('click', (e) => { const button = e.target.closest('.task-action-btn'); if (!button) return; const taskId = button.dataset.id; const currentTasks = AppStore.getState().tasks; const updatedTasks = currentTasks.map(task => task.id === taskId ? { ...task, done: !task.done } : task ); AppStore.setState({ tasks: updatedTasks }); }); }, /** * RECONCILIATION ENGINE (Bizim Mini React Fiber Algoritmamız) * DOM'u tamamen yıkmak yerine çocuk elementleri tarar, ekler, siler veya günceller. */ reconcile: function (state) { // 1. Statik Metin ve Progress Güncellemeleri const progress = EducationEngine.calculateProgress(state.tasks); if (this.progressBar) this.progressBar.style.width = `${progress}%`; if (this.progressText) this.progressText.textContent = `📊 İlerleme Durumu: %${progress}`; if (this.strategyText) this.strategyText.innerHTML = `🧭 Günlük Strateji: ${NarrativeManager.getStrategyText(progress)}`; // 2. Liste Boşsa Doğrudan Temizle if (state.tasks.length === 0) { this.taskListContainer.innerHTML = `

Aktif plan bulunamadı. Yukarıdan yeni bir döngü başlatın.

`; return; } // Boş durum mesajını temizle const emptyState = this.taskListContainer.querySelector('.empty-state'); if (emptyState) emptyState.remove(); // 3. RECONCILE LOOP: State listesi ile mevcut DOM listesini eşleştirme (Diffing algoritması) const existingNodes = Array.from(this.taskListContainer.children); state.tasks.forEach((task) => { const matchedNode = existingNodes.find(node => node.getAttribute('data-id') === task.id); if (matchedNode) { // Element zaten var -> Sadece değişen yerlerini akıllıca güncelle (SMART DIFF) TaskComponent.updateNode(matchedNode, task); } else { // Element yok -> Yeni oluştur ve DOM'a ekle (Append) const newNode = TaskComponent.createNode(task); this.taskListContainer.appendChild(newNode); } }); // State'te artık olmayan ama DOM'da kalan eski elementleri temizle (Garbage Collection) existingNodes.forEach(node => { const nodeId = node.getAttribute('data-id'); const stillExists = state.tasks.some(task => task.id === nodeId); if (!stillExists && nodeId) { node.remove(); } }); } }; // Sistemi güvenle başlat document.addEventListener('DOMContentLoaded', () => EducationController.init());
